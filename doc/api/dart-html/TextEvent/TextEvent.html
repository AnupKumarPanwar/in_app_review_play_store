factory TextEvent(String type,
    {bool canBubble: false,
    bool cancelable: false,
    Window? view,
    String? data}) {
  if (view == null) {
    view = window;
  }
  TextEvent e = document._createEvent("TextEvent") as TextEvent;
  e._initTextEvent(type, canBubble, cancelable, view, data);
  return e;
}
